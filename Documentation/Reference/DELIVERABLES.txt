6. USAGE
================================================================================

# Recommended method (interactive from project root)
python run_pipeline.py
# Select: 1=Override (first run), 2=Append (daily updates)

# Alternative: Direct CLI from bond_pipeline directory
cd bond_pipeline
python pipeline.py -i "../Raw Data/" -m override
python pipeline.py -i "../Raw Data/" -m append

# Alternative: Python module
python -m bond_pipeline.pipeline -i "Raw Data/" -m append

# Read parquet files
import pandas as pd
df_hist = pd.read_parquet('bond_data/parquet/historical_bond_details.parquet')
df_univ = pd.read_parquet('bond_data/parquet/universe.parquet')
